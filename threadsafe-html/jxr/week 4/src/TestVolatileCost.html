34    public static double Mark7(String msg, IntToDouble f) {


35      int n = 10, count = 1, totalCount = 0;


36      double dummy = 0.0, runningTime = 0.0, st = 0.0, sst = 0.0;


37      do { 


38        count *= 2;


39        st = sst = 0.0;


40        for (int j=0; j<n; j++) {


41          Timer t = new Timer();


42          for (int i=0; i<count; i++) 


43            dummy += f.call(i);


44          runningTime = t.check();


45          double time = runningTime * 1e6 / count; // microseconds


46          st += time; 


47          sst += time * time;


48          totalCount += count;


49        }


50      } while (runningTime < 0.25 && count < Integer.MAX_VALUE/2);


51      double mean = st/n, sdev = Math.sqrt(sst/n - mean*mean);


52      System.out.printf(
"%-25s %15.1f us %10.2f %10d%n"
, msg, mean, sdev, count);


53      return dummy / totalCount;


54    }

72  class IntArray {


73    private int[] array; 


74  


75    public IntArray(int length) {


76      array = new int[length];


77      for (int i=0; i<length; i++)


78        array[i] = 2 * i + 1;


79    }


80  


81    public boolean isSorted() {


82      for (int i=1; i<array.length; i++)


83        if (array[i-1] > array[i])


84  	return false;


85      return true;


86    }


87  }
