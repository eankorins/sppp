16      public static int countFactors(int p) {


17          if (p < 2) 


18              return 0;


19          int factorCount = 1, k = 2;


20          while (p >= k * k) {


21              if ((p % k == 0) && isPrime(k)) {


22                  factorCount++;


23                  p /= k;


24              } else 


25              k++;


26          }


27          return factorCount;


28      }


29      private static boolean isPrime(int n) {


30          int k = 2;


31          while (k * k <= n && n % k != 0)


32              k++;


33          return n >= 2 && k * k > n;


34      }

37  interface Histogram {


38    public void increment(int item);


39    public int getCount(int item);


40    public int getSpan();


41  }


42  


43  class Histogram2 implements Histogram{


44    final int[] counts;


45    public Histogram2(int span) {


46      this.counts = new int[span];


47    }


48    public synchronized void increment(int item) {


49      counts[item] = counts[item] + 1;


50    }


51    public int getCount(int item) {


52      return counts[item];


53    }


54    public int getSpan() {


55      return counts.length;


56    }
