33      double x = 1.1 * (double)(i & 0xFF);


34       return x * x * x * x * x * x * x * x * x * x 


35            * x * x * x * x * x * x * x * x * x * x;


36    }

231   public static double Mark4() {


232     int n = 10;


233     int count = 100_000_000;


234     double dummy = 0.0;


235     double st = 0.0, sst = 0.0;


236     for (int j=0; j<n; j++) {


237       Timer t = new Timer();


238       for (int i=0; i<count; i++) 


239         dummy += multiply(i);


240       double time = t.check() * 1e9 / count;


241       st += time; 


242       sst += time * time;


243     }


244     double mean = st/n, sdev = Math.sqrt(sst/n - mean*mean);


245     System.out.printf(
"%6.1f ns +/- %6.3f%n"
, mean, sdev);


246     return dummy / n;


247   }

249   public static double Mark5() {


250     int n = 10, count = 1, totalCount = 0;


251     double dummy = 0.0, runningTime = 0.0, st = 0.0, sst = 0.0;


252     do {


253       count *= 2;


254       st = sst = 0.0;


255       for (int j=0; j<n; j++) {


256         Timer t = new Timer();


257         for (int i=0; i<count; i++) 


258           dummy += multiply(i);


259         runningTime = t.check();


260         double time = runningTime * 1e9 / count;


261         st += time; 


262         sst += time * time;


263 	totalCount += count;


264       }


265       double mean = st/n, sdev = Math.sqrt(sst/n - mean*mean);


266       System.out.printf(
"%6.1f ns +/- %8.2f %10d%n"
, mean, sdev, count);


267     } while (runningTime < 0.25 && count < Integer.MAX_VALUE/2);


268     return dummy / totalCount;


269   }

346   public static double Mark8Setup(String msg, String info, Benchmarkable f, 


347                                   int n, double minTime) {


348     int count = 1, totalCount = 0;


349     double dummy = 0.0, runningTime = 0.0, st = 0.0, sst = 0.0;


350     do { 


351       count *= 2;


352       st = sst = 0.0;


353       for (int j=0; j<n; j++) {


354         Timer t = new Timer();


355         for (int i=0; i<count; i++) {


356           t.pause();


357           f.setup();


358           t.play();


359           dummy += f.call(i);


360         }


361         runningTime = t.check();


362         double time = runningTime * 1e9 / count;


363         st += time; 


364         sst += time * time;


365 	totalCount += count;


366       }


367     } while (runningTime < minTime && count < Integer.MAX_VALUE/2);


368     double mean = st/n, sdev = Math.sqrt(sst/n - mean*mean);


369     System.out.printf(
"%-25s %s%15.1f ns %10.2f %10d%n"
, msg, info, mean, sdev, count);


370     return dummy / totalCount;


371   }
