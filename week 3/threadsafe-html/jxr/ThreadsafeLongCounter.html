4   import java.io.IOException;


5   


6   // From JSR 305 jar file jsr305-3.0.0.jar:


7   import javax.annotation.concurrent.GuardedBy;


8   


9   public class ThreadsafeLongCounter {


10    public static void main(String[] args) throws IOException {


11      final LongCounter lc = new LongCounter();


12      Thread t = new Thread(new Runnable() {


13  	public void run() {


14  	  while (true)		// Forever call increment


15  	    lc.increment();


16  	}


17        });


18      t.start();


19      System.out.println(
"Press Enter to get the current value:"
);


20      while (true) {


21        System.in.read();         // Wait for enter key


22        System.out.println(lc.get()); 


23      }


24    }


25  }


26  


27  class LongCounter {


28    @GuardedBy(
"this"
)


29    private long count = 0;


30    public synchronized void increment() {


31      count++;


32    }


33    public synchronized long get() { 


34      return count; 


35    }


36  }
