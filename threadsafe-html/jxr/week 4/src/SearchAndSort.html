4   public class SearchAndSort {


5     static int linearSearch(int x, int[] arr) {


6       int n = arr.length, i = 0;                             


7       while (i < n) 


8         if (arr[i] != x) 


9           i++;


10        else 


11          return i;                     


12      return -1;


13    }


14    


15    static int binarySearch(int x, int[] arr) {


16      int n = arr.length, a = 0, b = n-1;                 


17      while (a <= b) {                                 


18        int i = (a+b) / 2;


19        if (x < arr[i]) 


20          b = i-1;


21        else if (arr[i] < x) 


22          a = i+1;


23        else 


24          return i;                  


25      }                                 


26      return -1;


27    }

29    // Utility for sorting


30    private static void swap(int[] arr, int s, int t) {


31      int tmp = arr[s];  arr[s] = arr[t];  arr[t] = tmp;


32    }


33  


34    // Selection sort


35    public static void selsort(int[] arr) { 


36      int n = arr.length;


37      for (int i = 0; i < n; i++) {


38        int least = i;                                      


39        for (int j = i+1; j < n; j++) 


40          if (arr[j] < arr[least])


41            least = j;


42        swap(arr, i, least);


43      }


44    }

46    // Quicksort


47    private static void qsort(int[] arr, int a, int b) { 


48      // sort arr[a..b]


49      if (a < b) { 


50        int i = a, j = b;


51        int x = arr[(i+j) / 2];                


52        do {                                   


53          while (arr[i] < x) i++;              


54          while (arr[j] > x) j--; 


55          if (i <= j) {


56            swap(arr, i, j);


57            i++; j--;


58          }                                    


59        } while (i <= j);                      


60        qsort(arr, a, j);                      


61        qsort(arr, i, b);                      


62      }                                        


63    }


64  


65    public static void quicksort(int[] arr) {


66      qsort(arr, 0, arr.length-1);


67    }

69    // Heapsort


70    private static void heapify(int[] arr, int i, int k) {


71      // heapify node arr[i] in the tree arr[0..k]


72      int j = 2 * i + 1;                          


73      if (j <= k) {


74        if (j+1 <= k && arr[j] < arr[j+1])


75          j++;                                  


76        if (arr[i] < arr[j]) {


77          swap(arr, i, j);                    


78          heapify(arr, j, k);                 


79        }           


80      }                                         


81    }


82  


83    public static void heapsort(int[] arr) {


84      int n = arr.length;


85      for (int m=n/2; m >= 0; m--) 


86        heapify(arr, m, n-1);


87      for (int m=n-1; m >= 1; m--) { 


88        swap(arr, 0, m);           


89        heapify(arr, 0, m-1);      


90      }                            


91    }

93    public static int[] fillIntArray(int n) {


94      int [] arr = new int[n];


95      for (int i = 0; i < n; i++)


96        arr[i] = i;


97      return arr;


98    }

100   private static final java.util.Random rnd = new java.util.Random();


101 


102   public static void shuffle(int[] arr) {


103     for (int i = arr.length-1; i > 0; i--)


104       swap(arr, i, rnd.nextInt(i+1));


105   }
