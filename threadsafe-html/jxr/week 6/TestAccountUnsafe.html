5   // locking on them individually does not work.

7   // TransferA: Locks only on the from-Account and therefore may perform


8   // non-atomic read-then-set; this may lose updates.  


9    


10  // TransferB: Updates each account atomically but not does not make


11  // the transfer atomically, so allows observations ("audits") in the


12  // middle of a transfer, where the account balances do not sum to 5000.

14  import java.util.Random;


15  


16  public class TestAccountUnsafe {


17    public static void main(String[] args) {


18      final Account account1 = new Account(), account2 = new Account();


19      final Random rnd = new Random();


20      final int transfers = 2_000_000;


21      account1.deposit(3000); account2.deposit(2000);


22      Thread clerk1 = new Thread(new Runnable() {


23          public void run() { 


24            for (int i=0; i<transfers; i++) 


25              account1.transferA(account2, rnd.nextInt(10000));


26          }


27        });


28      Thread clerk2 = new Thread(new Runnable() {


29          public void run() { 


30            for (int i=0; i<transfers; i++) 


31              account2.transferA(account1, rnd.nextInt(10000));


32          }


33        });


34      clerk1.start(); clerk2.start();


35      // We occasionally print the account balances during the transfer:


36      for (int i=0; i<40; i++) {


37        try { Thread.sleep(10); } catch (InterruptedException exn) { }


38        System.out.println(account1.get() + account2.get());


39      }


40      // The auditor prints the account balance sum when the clerks are finished: 


41      try { clerk1.join(); clerk2.join(); } catch (InterruptedException exn) { }


42      System.out.printf(
"%nFinal sum is %d%n"
, account1.get() + account2.get());


43    }


44  }

47  class Account {


48    private long balance = 0;


49  


50    public synchronized void deposit(long amount) {


51      balance += amount;


52    }


53  


54    public synchronized long get() {


55      return balance;


56    }


57  


58    // This may lose updates


59    public synchronized void transferA(Account that, long amount) {


60      this.balance = this.balance - amount;


61      that.balance = that.balance + amount;


62    }


63  


64    // This (wrongly) allows observation in the middle of a transfer


65    public void transferB(Account that, long amount) {


66      this.deposit(-amount);


67      that.deposit(+amount);


68    }


69  }
