1   // Crude wall clock timing utility, measuring time in seconds


2      


3   public class Timer {


4     private long start, spent = 0;


5     public Timer() { play(); }


6     public double check() { return (System.nanoTime()-start+spent)/1e9; }


7     public void pause() { spent += System.nanoTime()-start; }


8     public void play() { start = System.nanoTime(); }


9   }
