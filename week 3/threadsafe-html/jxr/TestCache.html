80      long k = 2;


81      while (p >= k * k) {


82        if (p % k == 0) {


83  	factors.add(k);


84  	p /= k;


85        } else 


86  	k++;


87      }


88      // Now k * k > p and no number in 2..k divides p

91      for (int i=0; i<result.length; i++) 


92        result[i] = factors.get(i);


93      return result;

98  /**


99   * Initial cache attempt using HashMap and synchronization; 


100  * suffers from lack of parallelism due to coarse locking.  


101  * From Goetz p. 103


102  * @author Brian Goetz and Tim Peierls


103  */


104 class Memoizer1 <A, V> implements Computable<A, V> {


105   private final Map<A, V> cache = new HashMap<A, V>();


106   private final Computable<A, V> c;


107   


108   public Memoizer1(Computable<A, V> c) { this.c = c; }


109 


110   public synchronized V compute(A arg) throws InterruptedException {


111     V result = cache.get(arg);


112     if (result == null) {


113       result = c.compute(arg);


114       cache.put(arg, result);


115     }


116     return result;


117   }


118 }

144 /**


145  * Memoizer3 


146  * Create a Future and register in cache immediately.


147  * Calls: ft.run() -> eval.call() -> c.compute(arg)


148  * From Goetz p. 106


149  * @author Brian Goetz and Tim Peierls


150  */


151 class Memoizer3<A, V> implements Computable<A, V> {


152   private final Map<A, Future<V>> cache 


153     = new ConcurrentHashMap<A, Future<V>>();


154   private final Computable<A, V> c;


155   


156   public Memoizer3(Computable<A, V> c) { this.c = c; }


157   


158   public V compute(final A arg) throws InterruptedException {


159     Future<V> f = cache.get(arg);


160     if (f == null) {


161       Callable<V> eval = new Callable<V>() {


162 	  public V call() throws InterruptedException {


163 	    return c.compute(arg);


164       }};


165       FutureTask<V> ft = new FutureTask<V>(eval);


166       cache.put(arg, ft);


167       f = ft;


168       ft.run();


169     }


170     try { return f.get(); } 


171     catch (ExecutionException e) { throw launderThrowable(e.getCause()); }  


172   }


173 


174   public static RuntimeException launderThrowable(Throwable t) {


175     if (t instanceof RuntimeException)


176       return (RuntimeException) t;


177     else if (t instanceof Error)


178       throw (Error) t;


179     else


180       throw new IllegalStateException(
"Not unchecked"
, t);


181   }


182 }

185 /**


186  * Memoizer4, hybrid of variant of Goetz's Memoizer3 and Memoizer.  If


187  * arg not in cache, create Future, then atomically putIfAbsent in


188  * cache, then run on calling thread.


189  */


190 


191 class Memoizer4<A, V> implements Computable<A, V> {


192   private final Map<A, Future<V>> cache 


193     = new ConcurrentHashMap<A, Future<V>>();


194   private final Computable<A, V> c;


195   


196   public Memoizer4(Computable<A, V> c) { this.c = c; }


197   


198   public V compute(final A arg) throws InterruptedException {


199     Future<V> f = cache.get(arg);


200     if (f == null) {


201       Callable<V> eval = new Callable<V>() {


202 	  public V call() throws InterruptedException {


203 	    return c.compute(arg);


204       }};


205       FutureTask<V> ft = new FutureTask<V>(eval);


206       f = cache.putIfAbsent(arg, ft);


207       if (f == null) { 


208 	f = ft; 


209 	ft.run();


210       }


211     }


212     try { return f.get(); } 


213     catch (ExecutionException e) { throw launderThrowable(e.getCause()); }  


214   }


215 


216   public static RuntimeException launderThrowable(Throwable t) {


217     if (t instanceof RuntimeException)


218       return (RuntimeException) t;


219     else if (t instanceof Error)


220       throw (Error) t;


221     else


222       throw new IllegalStateException(
"Not unchecked"
, t);


223   }


224 }

226 /**


227  * Memoizer5, modern variant of Memoizer4 using the new Java 8


228  * computeIfAbsent.  Atomically test whether arg is in cache and if


229  * not create Future and put it there, then run on calling thread.


230  */


231 


232 class Memoizer5<A, V> implements Computable<A, V> {


233   private final Map<A, Future<V>> cache 


234     = new ConcurrentHashMap<A, Future<V>>();


235   private final Computable<A, V> c;


236   


237   public Memoizer5(Computable<A, V> c) { this.c = c; }


238   


239   public V compute(final A arg) throws InterruptedException {


240     // This seems gross, I just need to assign a variable outside the Function...


241     final AtomicReference<FutureTask<V>> ftr = new AtomicReference<FutureTask<V>>();


242     Future<V> f = cache.computeIfAbsent(arg, new Function<A,Future<V>>() { 


243 	public Future<V> apply(final A arg) {


244 	  Callable<V> eval = new Callable<V>() {


245 	      public V call() throws InterruptedException {


246 		return c.compute(arg);


247 	      }};


248 	  ftr.set(new FutureTask<V>(eval));


249 	  return ftr.get();


250 	}});


251     // Important to run() the future outside the computeIfAbsent():


252     if (ftr.get() != null) 


253       ftr.get().run();


254     try { return f.get(); } 


255     catch (ExecutionException e) { throw launderThrowable(e.getCause()); }  


256   }


257 


258   public static RuntimeException launderThrowable(Throwable t) {


259     if (t instanceof RuntimeException)


260       return (RuntimeException) t;


261     else if (t instanceof Error)


262       throw (Error) t;


263     else


264       throw new IllegalStateException(
"Not unchecked"
, t);


265   }


266 }

269 /**


270  * Final implementation of Memoizer using cheap get() followed by


271  * atomic putIfAbsent.


272  * From Goetz p. 108


273  * @author Brian Goetz and Tim Peierls


274  */


275 class Memoizer <A, V> implements Computable<A, V> {


276   private final ConcurrentMap<A, Future<V>> cache


277     = new ConcurrentHashMap<A, Future<V>>();


278   private final Computable<A, V> c;


279   


280   public Memoizer(Computable<A, V> c) { this.c = c; }


281 


282   public V compute(final A arg) throws InterruptedException {


283     while (true) {


284       Future<V> f = cache.get(arg);


285       if (f == null) {


286 	Callable<V> eval = new Callable<V>() {


287 	    public V call() throws InterruptedException {


288 	      return c.compute(arg);


289 	    }


290 	  };


291 	FutureTask<V> ft = new FutureTask<V>(eval);


292 	f = cache.putIfAbsent(arg, ft);


293 	if (f == null) {


294 	  f = ft;


295 	  ft.run();


296 	}


297       }


298       try {


299 	return f.get();


300       } catch (CancellationException e) {


301 	cache.remove(arg, f);


302       } catch (ExecutionException e) {


303 	throw launderThrowable(e.getCause());


304       }


305     }


306   }

309   /**


310    * Coerce a checked Throwable to an unchecked RuntimeException.


311    


312    * sestoft@itu.dk 2014-09-07: This method converts a Throwable


313    * (which is a checked exception) into a RuntimeException (which is


314    * an unchecked exception) or an IllegalStateException (which is a


315    * subclass of RuntimeException and hence unchecked).  It is unclear


316    * why RuntimeException and Error are treated differently; both are


317    * unchecked.  A simpler (but grosser) approach is to simply throw a


318    * new RuntimeException(t), thus wrapping the Throwable, but that


319    * may lead to a RuntimeException containing a RuntimeException


320    * which is a little strange.  The original


321    * java.util.concurrent.ExecutionException that wrapped the


322    * Throwable is itself checked and therefore needs to be caught and


323    * turned into something less obnoxious.


324 


325    * @author Brian Goetz and Tim Peierls 


326    */


327 


328   public static RuntimeException launderThrowable(Throwable t) {


329     if (t instanceof RuntimeException)


330       return (RuntimeException) t;


331     else if (t instanceof Error)


332       throw (Error) t;


333     else


334       throw new IllegalStateException(
"Not unchecked"
, t);


335   }
