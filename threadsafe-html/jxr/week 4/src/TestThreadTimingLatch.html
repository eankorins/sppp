20    /**


21     * Using CountDownLatch for starting and stopping threads in timing tests.


22     * @author Brian Goetz and Tim Peierls; modifications by sestoft@itu.dk


23     */


24    


25    public static double timeTasks(int threadCount, final Runnable task)


26      throws InterruptedException {


27      final CountDownLatch startGate = new CountDownLatch(1);


28      final CountDownLatch endGate = new CountDownLatch(threadCount);


29      for (int i = 0; i < threadCount; i++) {


30        Thread t = new Thread(new Runnable() { public void run() {


31  	try {


32  	  startGate.await();


33  	  try { task.run(); } 


34  	  finally { endGate.countDown(); }


35  	} catch (InterruptedException ignored) { }


36        } } );


37        t.start();


38      }


39  


40      Timer timer = new Timer();


41      startGate.countDown();


42      endGate.await();


43      return timer.check();


44    }


45  }

47  interface Factorizer {


48    public long[] getFactors(long p);


49    public long getCount();


50  }


51  


52  // Like Goetz p. 31.


53  class CachingFactorizer implements Factorizer {


54    private long lastNumber = 1;


55    private long[] lastFactors = new long[0];


56  


57    public long[] getFactors(long p) {


58      long[] factors = null;


59      synchronized (this) {


60        if (p == lastNumber)


61  	factors = lastFactors.clone();


62      }


63      if (factors == null) {


64        factors = PrimeFactors.compute(p);


65        synchronized (this) {


66  	lastNumber = p;


67  	lastFactors = factors.clone();


68        }


69      }


70      return factors;


71    }


72    public long getCount() { return 0; }


73  }

75  class PrimeFactors {


76    public static long[] compute(long p) {


77      ArrayList<Long> factors = new ArrayList<Long>();


78      long k = 2;


79      while (p >= k * k) {


80        if (p % k == 0) {


81  	factors.add(k);


82  	p /= k;


83        } else 


84  	k++;


85      }


86      // Now k * k > p and no number in 2..k divides p


87      factors.add(p);


88      long[] result = new long[factors.size()];


89      for (int i=0; i<result.length; i++) 


90        result[i] = factors.get(i);


91      return result;


92    }


93  


94    public static boolean check(long p, long[] factors) {


95      long prod = 1;


96      for (int i=0; i<factors.length; i++)


97        prod *= factors[i];


98      return p == prod;


99    }


100 }

104 class Timer {


105   private long start, spent = 0;


106   public Timer() { play(); }


107   public double check() { return (System.nanoTime()-start+spent)/1e9; }


108   public void pause() { spent += System.nanoTime()-start; }


109   public void play() { start = System.nanoTime(); }
